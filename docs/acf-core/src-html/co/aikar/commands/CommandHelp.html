026import com.google.common.collect.SetMultimap;

036@SuppressWarnings("WeakerAccess")
037public class CommandHelp {
038    private final CommandManager manager;
039    private final CommandIssuer issuer;
040    private final List<HelpEntry> helpEntries = new ArrayList<>();
041    private final String commandName;
042    final String commandPrefix;
043    private int page = 1;
044    private int perPage;
045    List<String> search;
046    private Set<HelpEntry> selectedEntry = new HashSet<>();
047    private int totalResults;
048    private int totalPages;
049    private boolean lastPage;
050
051    public CommandHelp(CommandManager manager, RootCommand rootCommand, CommandIssuer issuer) {
052        this.manager = manager;
053        this.issuer = issuer;
054        this.perPage = manager.defaultHelpPerPage;
055        this.commandPrefix = manager.getCommandPrefix(issuer);
056        this.commandName = rootCommand.getCommandName();
057
058
059        SetMultimap<String, RegisteredCommand> subCommands = rootCommand.getSubCommands();
060        Set<RegisteredCommand> seen = new HashSet<>();
061        
062        if (!rootCommand.getDefCommand().hasHelpCommand) {
063            RegisteredCommand defCommand = rootCommand.getDefaultRegisteredCommand();
064            if (defCommand != null) {
065                helpEntries.add(new HelpEntry(this, defCommand));
066                seen.add(defCommand);
067            }
068        }
069        
070        subCommands.entries().forEach(e -> {
071            String key = e.getKey();
072            if (key.equals(BaseCommand.DEFAULT) || key.equals(BaseCommand.CATCHUNKNOWN)) {
073                return;
074            }
075
076            RegisteredCommand regCommand = e.getValue();
077
078            if (!regCommand.isPrivate && regCommand.hasPermission(issuer) && !seen.contains(regCommand)) {
079                this.helpEntries.add(new HelpEntry(this, regCommand));
080                seen.add(regCommand);
081            }
082        });
083    }
084
085    @UnstableAPI // Not sure on this one yet even when API becomes unstable
086    protected void updateSearchScore(HelpEntry help) {
087        if (this.search == null || this.search.isEmpty()) {
088            help.setSearchScore(1);
089            return;
090        }
091        final RegisteredCommand<?> cmd = help.getRegisteredCommand();
092
093        int searchScore = 0;
094        for (String word : this.search) {
095            Pattern pattern = Pattern.compile(".*" + Pattern.quote(word) + ".*", Pattern.CASE_INSENSITIVE);
096            for (String subCmd : cmd.registeredSubcommands) {
097                Pattern subCmdPattern = Pattern.compile(".*" + Pattern.quote(subCmd) + ".*", Pattern.CASE_INSENSITIVE);
098                if (pattern.matcher(subCmd).matches()) {
099                    searchScore += 3;
100                } else if (subCmdPattern.matcher(word).matches()) {
101                    searchScore++;
102                }
103            }
104
105
106            if (pattern.matcher(help.getDescription()).matches()) {
107                searchScore += 2;
108            }
109            if (pattern.matcher(help.getParameterSyntax()).matches()) {
110                searchScore++;
111            }
112            if (help.getSearchTags() != null && pattern.matcher(help.getSearchTags()).matches()) {
113                searchScore += 2;
114            }
115        }
116        help.setSearchScore(searchScore);
117    }
118
119    public CommandManager getManager() {
120        return manager;
121    }
122
123    public boolean testExactMatch(String command) {
124        selectedEntry.clear();
125        for (HelpEntry helpEntry : helpEntries) {
126            if (helpEntry.getCommand().endsWith(" " + command)) {
127                selectedEntry.add(helpEntry);
128            }
129        }
130        return !selectedEntry.isEmpty();
131    }
132
133    public void showHelp() {
134        showHelp(issuer);
135    }
136
137    public void showHelp(CommandIssuer issuer) {
138        CommandHelpFormatter formatter = manager.getHelpFormatter();
139        if (!selectedEntry.isEmpty()) {
140            HelpEntry first = ACFUtil.getFirstElement(selectedEntry);
141            formatter.printDetailedHelpHeader(this, issuer, first);
142
143            for (HelpEntry helpEntry : selectedEntry) {
144                formatter.showDetailedHelp(this, helpEntry);
145            }
146
147            formatter.printDetailedHelpFooter(this, issuer, first);
148            return;
149        }
150
151        List<HelpEntry> helpEntries = getHelpEntries();
152        Iterator<HelpEntry> results = helpEntries.stream()
153                .filter(HelpEntry::shouldShow)
154                .sorted(Comparator.comparingInt(helpEntry -> helpEntry.getSearchScore() * -1)).iterator();
155        if (!results.hasNext()) {
156            issuer.sendMessage(MessageType.ERROR, MessageKeys.NO_COMMAND_MATCHED_SEARCH, "{search}", ACFUtil.join(this.search, " "));
157            helpEntries = getHelpEntries();
158            results = helpEntries.iterator();
159        }
160        this.totalResults = helpEntries.size();
161        int min = (this.page - 1) * this.perPage; // TODO: per page configurable?
162        int max = min + this.perPage;
163        this.totalPages = (int) Math.ceil((float) totalResults / (float) this.perPage);
164        int i = 0;
165        if (min >= totalResults) {
166            issuer.sendMessage(MessageType.HELP, MessageKeys.HELP_NO_RESULTS);
167            return;
168        }
169
170        List<HelpEntry> printEntries = new ArrayList<>();
171        while (results.hasNext()) {
172            HelpEntry e = results.next();
173            if (i >= max) {
174                break;
175            }
176            if (i++ < min) {
177                continue;
178            }
179            printEntries.add(e);
180        }
181        this.lastPage = max >= totalResults;
182
183        if (search == null) {
184            formatter.showAllResults(this, printEntries);
185        } else {
186            formatter.showSearchResults(this, printEntries);
187        }
188
189    }

192        return helpEntries;
193    }

206    }

208    public void setSearch(List<String> search) {
209        this.search = search;
210        getHelpEntries().forEach(this::updateSearchScore);
211    }
